int* performOps(int *A, int len, int *blen) {

int i;

*blen = len * 2;

int *B = (int *)malloc((*blen) * sizeof(int));

for (i = 0; i < len; i++) {

B[i] = A[i];

B[i + len] = A[(len - i) % len];

}

return B;

}
